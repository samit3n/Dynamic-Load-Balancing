272 unsigned TileDescriptor::getOverlapOffset(const TileDescriptor & tile, unsigned & cnt ) const
273 {
274  Dims tpos = tile.getPosition(); // absolute position
275  Dims tsize = tile.getSize();
276  Dims tend(tpos.x + tsize.x , tpos.y + tsize.y ); // absolute tile end
277 
278  Dims end(position.x + size.x, position.y + size.y); //absolute end
279 
280  TEdge shared = getSharedEdge(tile);
281  unsigned offset = 100;
282 
283  // offset is computed based on which edge is shared and
284  // whether the neighbor tile begins of ends on the edge.
285 
286  // cout << "shared " << shared << endl;
287 
288  if(shared == TOP){
289  // cout << "top" << endl;
290 
291  // neighbor begins and ends inside
292  if(tpos.x >= position.x && tend.x <= end.x){
293 
294  offset = tpos.x - position.x; // (end.x - tend.x);
295  cnt = tend.x - tpos.x;
296 
297  // begins inside, ends somewheres
298  }else if(tpos.x >= position.x && tend.x > end.x){
299 
300  offset = tpos.x - position.x;
301  cnt = end.x - tpos.x;
302 
303  // begins somewhere, end inside
304  }else if(tpos.x < position.x && tend.x <= end.x){
305 
306  offset = 0;
307  cnt = tend.x - position.x;
308 
309  }else{ // whole border
310 
311  offset = 0;
312  cnt = size.x;
313  }
314 
315  }else if(shared == RIGHT){
316  // cout << "right" << endl;
317 
318  // neighbor begins and ends inside
319  if(tpos.y >= position.y && tend.y <= end.y){
320 
321  offset = size.x + ( tpos.y - position.y);
322  cnt = tend.y - tpos.y;
323 
324  // begins inside, ends somewhere
325  }else if(tpos.y >= position.y && tend.y > end.y){
326 
327  offset = size.x + (tpos.y - position.y);
328  cnt = end.y - tpos.y;
329 
330  // begins somewhere, end inside
331  }else if(tpos.y < position.y && tend.y <= end.y ){
332 
333  offset = size.x;
334  cnt = tend.y - position.y;
335 
336  // whole side
337  }else{
338 
339  offset = size.x;
340  cnt = size.y;
341  }
342 
343 
344  }else if(shared == BOTTOM){
345  // cout << "bottom" << endl;
346 
347 
348  if(tpos.x >= position.x && tend.x <= end.x){
349  offset = size.y + (tpos.x - position.x); // (end.x - tend.x);
350  cnt = tend.x - tpos.x;
351 
352  }else if(tpos.x >= position.x && tend.x > end.x){
353 
354  offset = size.y + (tpos.x - position.x);
355  cnt = end.x - tpos.x;
356 
357  }else if(tpos.x < position.x && tend.x <= end.x){
358 
359  offset = size.y;
360  cnt = tend.x - position.x;
361  }else{ // whole border
362 
363  offset = size.y;
364  cnt = size.x;
365  }
366 
367  // add offset of TOP and BOTTOM in buffer
368  offset += size.x + size.y;
369 
370  }else if(shared == LEFT){
371  // cout << "left" << endl;
372 
373  if(tpos.y >= position.y && tend.y <= end.y){
374 
375  offset = tpos.y - position.y;
376  cnt = tend.y - tpos.y;
377 
378  }else if(tpos.y >= position.y && tend.y > end.y){
379 
380  offset = tpos.y - position.y;
381  cnt = end.y - tpos.y;
382 
383  }else if(tpos.y < position.y && tend.y <= end.y ){
384 
385  offset = 0;
386  cnt = tend.y - position.y;
387 
388  }else{
389 
390  offset = 0;
391  cnt = size.y;
392  }
393 
394  // add offset of TOP and BOTTOM in buffer
395  offset += size.x + size.y;
396 
397  }else{
398 
399  throw std::runtime_error("TileDescriptor::getOverlapOffset error, tile is not neighbor");
400  }
401
