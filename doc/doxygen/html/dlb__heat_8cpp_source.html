312 void ComputePoint(float *oldTemp,
313  float *newTemp,
314  float *params,
315  int *map,
316  size_t i,
317  size_t j,
318  size_t edgeSize,
319  float airFlowRate,
320  float coolerTemp)
321 {
322  // [i] Calculate neighbor indices
323  const int center = i * edgeSize + j;
324  const int top[2] = { center - (int)edgeSize, center - 2*(int)edgeSize };
325  const int bottom[2] = { center + (int)edgeSize, center + 2*(int)edgeSize };
326  const int left[2] = { center - 1, center - 2};
327  const int right[2] = { center + 1, center + 2};
328 
329  // [ii] The reciprocal value of the sum of domain parameters for normalization
330  const float frac = 1.0f / (params[top[0]] + params[top[1]] +
331  params[bottom[0]] + params[bottom[1]] +
332  params[left[0]] + params[left[1]] +
333  params[right[0]] + params[right[1]] +
334  params[center]);
335 
336  // [iii] Calculate new temperature in the grid point
337  float pointTemp =
338  oldTemp[top[0]] * params[top[0]] * frac +
339  oldTemp[top[1]] * params[top[1]] * frac +
340  oldTemp[bottom[0]] * params[bottom[0]] * frac +
341  oldTemp[bottom[1]] * params[bottom[1]] * frac +
342  oldTemp[left[0]] * params[left[0]] * frac +
343  oldTemp[left[1]] * params[left[1]] * frac +
344  oldTemp[right[0]] * params[right[0]] * frac +
345  oldTemp[right[1]] * params[right[1]] * frac +
346  oldTemp[center] * params[center] * frac;
347 
348  // [iv] Remove some of the heat due to air flow (5% of the new air)
349  pointTemp = (map[center] == 0)
350  ? (airFlowRate * coolerTemp) + ((1.0f - airFlowRate) * pointTemp)
351  : pointTemp;
352 
353  newTemp[center] = pointTemp;
354 }
