183 void TopologyDescriptor::countDispls(void)
184 {
185  unsigned cnt;
186  Neighbor nbor;
187 
188  for(auto n: neighbors){
189  // find neighbor interator by rank
190  auto res = find(tiles.begin(), tiles.end(), n);
191  // get neighbor index
192  unsigned idx = res - tiles.begin();
193  unsigned dsp = myTile.getOverlapOffset(tiles[idx], cnt);
194 
195  nbor.displ = 2*dsp;
196  nbor.count = 2*cnt;
197  nbor.wRank = n;
198 
199  displs.push_back(2*dsp);
200  counts.push_back(2*cnt);
201 
202  nData.insert(pair<int, Neighbor>(n, nbor ));
203  // nbor.clear();
204 
205  }
206 
207  // insert counts and dipls for scatter for this tile
208  // must be put to appropriate place relative to neighbors
209  bool once = true;
210  for(auto it = neighbors.begin(); it < neighbors.end();it++){
211 
212  if(once && (*neighbors.begin()) > rank){
213 
214  counts.insert(counts.begin(), 0);
215  displs.insert(displs.begin(), 0);
216  once = false;
217 
218  }else if(once && *(neighbors.end() - 1) < rank){
219 
220  counts.insert(counts.end(), 0);
221  displs.insert(displs.end(), 0);
222  once = false;
223 
224  }else if(once && (*it < rank) && ((it+1) != neighbors.end()) && (*(it+ 1) > rank) )
225  { //somewhere inside
226  unsigned idx = it - neighbors.begin();
227  counts.insert(counts.begin() + idx +1, 0);
228  displs.insert(displs.begin() + idx +1, 0);
229  once = false;
230  }
231  }
232 }

247  //iterate over ranks

273

276  myComm = newcomm;

286  MPI_assert( MPI_Bcast(scatDisp, size, MPI_INT, myCommRank, myComm) LOCATION);

289  delete[] scatDisp;
290 
291 
292 
293  }else{ // i is neighbor
294

300  nData.at(i).myRank = tmp;
301 
302  MPI_assert( MPI_Comm_group(newcomm, (&tmpGroup)), "tmpGroup failed" LOCATION);

304 
305  if(root == MPI_UNDEFINED)
306  throw runtime_error("rank translation failed");
307  // since I don't know it's rank, I have to use ANY_SOURCE
308  nData.at(i).root = root;
309 
310 
311  MPI_assert( MPI_Bcast(scatCnts, size, MPI_INT, root, newcomm) LOCATION);
312  MPI_assert( MPI_Bcast(scatDisp, size, MPI_INT, root, newcomm) LOCATION);
313 
314  nData.at(i).scatterCnts = scatCnts;
315  nData.at(i).scatterDispls = scatDisp;

317  }
318 
319  }else{
320 
321  // do not participate in this communicator
322  // I'm not actually on turn and i is no my neighbour
323  // will receive MPI_COMM_NULL
